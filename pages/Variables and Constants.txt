// Functions

func greet(person: String) -> String {
    return "Hello, \(person)!"
}

let greetingMessage = greet(person: "Bob")
print(greetingMessage)

// Classes and Structures

class Person {
    var name: String
    var age: Int
    
    init(name: String, age: Int) {
        self.name = name
        self.age = age
    }
    
    func description() -> String {
        return "\(name) is \(age) years old."
    }
}

struct Point {
    var x: Int
    var y: Int
}

// Usage of Classes and Structures

let person1 = Person(name: "John", age: 30)
let person2 = Person(name: "Jane", age: 25)

print(person1.description())
print(person2.description())

let point1 = Point(x: 5, y: 10)
let point2 = Point(x: 3, y: 7)

print("Point 1: (\(point1.x), \(point1.y))")
print("Point 2: (\(point2.x), \(point2.y))")
